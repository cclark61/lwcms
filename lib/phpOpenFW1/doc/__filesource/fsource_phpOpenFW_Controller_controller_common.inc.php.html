  91. function reg_data_source($ds_index, $ds_params)
  92. {
  93.     $known_params = array('type', 'server', 'port', 'source', 'user', 'pass', 'instance', 'conn_str', 'options', 'persistent', 'reuse_connection');
  94.     $optional_params = array('port' => '', 'user' => '', 'pass' => '', 'instance' => '', 'conn_str' => '', 'options' => '', 'persistent' => '', 'reuse_connection' => '');
  95.     settype($ds_index, 'string');
  96.     if (!is_array($ds_params)) {
  97.         trigger_error("Error: reg_data_source(): Index: '{$ds_index}', Second parameter must be an array.");
  98.         return 2;
  99.     }
  100.     else {
  101.         $param_count = count($known_params);
  102.         $new_data_source = array();
  103.         foreach ($known_params as $param_index) {
  104.             if (isset($ds_params[$param_index])) {
  105.                 $new_data_source[$param_index] = $ds_params[$param_index];
  106.                 $param_count--;
  107.             }
  108.             else if (isset($optional_params[$param_index])) { $param_count--; }
  109.         }
  110.  
  111.         if ($param_count > 0) {
  112.             trigger_error("Error: reg_data_source(): Index: '{$ds_index}', Incorrect parameter count in parameter array.");
  113.             return 4;
  114.         }
  115.         else {
  116.             $_SESSION[$ds_index] = $new_data_source;
  117.             return 0;
  118.         }
  119.     }
  120. }
