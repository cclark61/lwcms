  84.     // Render Function
  85.     //*************************************************************************
  86.     //*************************************************************************
  87.     public function render($buffer=false)
  88.     {
  89.         if ($buffer) { ob_start(); }
  90.         $el_cont = '';
  91.  
  92.         // Tabs
  93.         $tabs = '';
  94.         for ($i = 0; $i < $this->tabs; $i++) { $tabs .= "\t"; }
  95.  
  96.         //***************************************
  97.         // Child Nodes OR Inset Text
  98.         //***************************************
  99.         if ($this->child_nodes) {
  100.             // Child Nodes
  101.             foreach ($this->child_nodes as $node) {
  102.                 if (is_object($node) && get_class($node) == 'gen_element') {
  103.                      $node->set_tabs($this->tabs + 1);
  104.                 }
  105.                 $el_cont .= $node;
  106.             }
  107.         }
  108.         else if ($this->inset_val !== false && $this->inset_val !== '') {
  109.             // Inset Value
  110.             $el_cont .= ($this->style == 'tree') ? ("\t$tabs" . $this->inset_val . "\n") : ($this->inset_val);
  111.         }
  112.  
  113.         //***************************************
  114.         // Opening Tag        
  115.         //***************************************
  116.         $this->element_xml = "$tabs<$this->element";
  117.         if (isset($this->attributes)) {
  118.             foreach($this->attributes as $key=>$value) {
  119.                 $this->element_xml .= " $key=\"$value\"";
  120.             }
  121.         }
  122.  
  123.         //***************************************
  124.         // Build Closing Tag
  125.         //***************************************
  126.         $close_tag = "</$this->element>\n";
  127.         if ($this->style == 'tree') { $close_tag = $tabs . $close_tag; }
  128.  
  129.         //***************************************
  130.         // Content / Closing Tag
  131.         //***************************************
  132.         if ($el_cont !== false && $el_cont !== '') {
  133.             if ($this->style == 'tree') {
  134.                 $this->element_xml .= ">\n" . $el_cont . $close_tag;    
  135.             }
  136.             else { $this->element_xml .= '>' . $el_cont . $close_tag; }
  137.  
  138.         }
  139.         else if ($this->endtag) { $this->element_xml .= ">$close_tag"; }
  140.         else { $this->element_xml .=  " />\n"; }
  141.  
  142.         //***************************************
  143.         // Perform XML Transformation
  144.         //***************************************
  145.         $sxoe = (isset($_SESSION['show_xml_on_error']) && $_SESSION['show_xml_on_error'] == 1) ? (true) : (false);
  146.         xml_transform($this->element_xml, $this->xsl_template, $sxoe);
  147.         if ($buffer) { return ob_get_clean(); }
  148.     }

  172.     /**
  173.     * Add, append, overwrite, or remove an attribute value to the element

  175.     * @param string Attribute value

  177.     //*************************************************************************
  178.     // Add an attribute
  179.     //*************************************************************************
  180.     public function attr($attr, $value=false, $append=true)
  181.     {
  182.         $key = (string)$attr;
  183.         $val = (string)$value;
  184.         $append = (bool)$append;
  185.         $key_exists = (isset($this->attributes[$key])) ? (true) : (false);
  186.         if ($key == '') { return false; }
  187.         if ($val == '') {
  188.             if (isset($this->attributes[$key])) {
  189.                 unset($this->attributes[$key]);
  190.             }
  191.             else { return false; }
  192.         }
  193.         else {
  194.             if ($append && $key_exists) {
  195.                 $this->attributes[$key] .= ' ' . $value;
  196.             }
  197.             else {
  198.                 $this->attributes[$key] = $value;
  199.             }
  200.         }
  201.         
  202.         return true;
  203.     }

  214.     public function attrs($attrs, $append=true)
  215.     {
  216.         if (!is_array($attrs)) { return false; }

  218.         $count = 0;
  219.         foreach ($attrs as $attr => $value) {
  220.             if ($this->attr($attr, $value, $append)) { $count++; }
  221.         }
  222.  
  223.         return $count;
  224.     }

  316. /**
  317. * Creates a generic xml element with content
  318. * @package        phpOpenFW
  319. * @subpackage    Objects
  320. * @param string Element Name (ie. "div", "data", "p", etc.) Can be XML or XHTML
  321. * @param string Content inside of element
  322. * @param array An array, in the form of [key] => [value], of attributes
  323. */
  324. //*****************************************************************************
  325. class gen_element extends element
  326. {
  327.     public function __construct($element, $content=false, $attrs=false)
  328.     {
  329.         $this->element = $element;
  330.         $this->tabs = 0;
  331.         if ($content !== false && $content !== '') { $this->inset_val = $content; }
  332.         if (is_array($attrs)) {
  333.             foreach ($attrs as $key => $val) { $this->set_attribute($key, $val); }
  334.         }
  335.     }
  336. }

  345. function xhe($elm=false, $content='', $attrs=array(), $escape=false) {
  346.     if ($elm) {
  347.         ob_start(); 
  348.         $c = new gen_element($elm, $content, $attrs);

  350.         return ($escape) ? (xml_escape(ob_get_clean())) : (ob_get_clean());
  351.     }
  352.     return false;
  353. }
